# _plugins/generate_tags.rb module Jekyll class TagPageGenerator < Generator safe true def generate(site) site.tags.each do |tag, posts| slug = tag.downcase.strip.gsub(' ', '-').gsub(/[^\w-]/, '') site.pages << TagPage.new(site, site.source, File.join('tags', slug), tag) end end end class TagPage < Page def initialize(site, base, dir, tag) @site = site @base = base @dir = dir @name = 'index.html' self.process(@name) self.read_yaml(File.join(base, '_layouts'), 'tag.html') self.data['tag'] = tag self.data['title'] = "Posts tagged: #{tag}" end end end
